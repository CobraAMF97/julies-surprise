function checkLocation() { if (navigator.geolocation) { navigator.geolocation.getCurrentPosition(function(position) { var lat = position.coords.latitude; var lon = position.coords.longitude; // Coordonnées approximatives de Trouville-sur-Mer var trouvilleLat = 49.3667; var trouvilleLon = 0.0833; // Tolérance de distance réduite var tolerance = 0.05; if (Math.abs(lat - trouvilleLat) <= tolerance && Math.abs(lon - trouvilleLon) <= tolerance) { document.getElementById("hidden-content").style.display = "block"; } else { alert("Cette page est actuellement inaccessible. Revenez plus tard."); } }); } else { alert("La géolocalisation n'est pas supportée par ce navigateur."); } }
